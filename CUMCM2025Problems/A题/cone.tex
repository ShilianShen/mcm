\usepackage{tikz}
\usetikzlibrary{arrows.meta}
\begin{document}
\begin{tikzpicture}[>=Stealth, scale=0.8]
    % 坐标轴
    \draw[->] (-1,0) -- (5,0) node[right] {$x$};
    \draw[->] (0,-1) -- (0,5) node[above] {$y$};

    % 网格（可选）
    \draw[gray!20] (-1,-1) grid (5,5);

    % 向量
    \draw[->, ultra thick, blue] (0,0) -- (4,2)
        node[midway, above left] {$\vec{v} = (4,2)$};

    % 向量分量
    \draw[->, thick, red, dashed] (0,0) -- (4,0) node[below] {$v_x$};
    \draw[->, thick, green!70!black, dashed] (4,0) -- (4,2) node[right] {$v_y$};

    % 原点标记
    \node at (0,0) [below left] {O};
